# 1 конфигурация глобальное имя
git config --global user.name "Pavel Shift"
# 2 конфигурация почта
git config --global user.mail "shift.pavel@mail.ru"

# 3 инициализация
git init

# история коммитов
git log
# + изменений
git log -p

# 4 статус файлов
git status

# 5 добавление файла под версионный контроль
git add git.txt

# 6 коммит
git commit -m "Старт проекта"
# + взять все файлы под версионным контролем
git commit -am "Старт проекта"


# показывает подробно
git show

# возврашение к последнему коммиту
git restore

# показывает разницу в файле с последнего коммита
git diff
# + в индексе
git diff --staged

# переименование или перемещение файлов
git mv file1 file2
# удаление из идекса
git rm file1

# 7 игнорирование файлов 
-делаем файл .gitignore

# Смена пользователя
git config --global credential.username "Pavel Shift"

# Отправка изменнеий в репозиторий
git push -u origin main

# Показ существующих веток
git branch

# Новая ветка
git branch new

# Переход на ветку
git checkout new
